B) Total:(2(n-2)(n-2))+(n(n-2))+n+1= 3n^2-10n+9

for i <- 0 to n-1 do              // n
    for j <- 0 to n-1 do          // n(n-2) = n^2-2n
        sum <- sum + 1            // 2(n-2)(n-2) = 2n^2-8n+8
